# useInterval

This is how I apply the `useInterval` hook.

```jsx
import React, { useState } from 'react'
import { useInterval } from '@daneroo/qcic-react'

export function Counter ({ delay = 1000 }) {
  const [count, setCount] = useState(0)

  useInterval(() => {
    setCount(count => count + 1)
  }, delay)

  return (
    <div>
      <h4>Counter: {count}</h4>
    </div>
  )
}
```

This doesn't work yet; the Fetch is not re-evaluated.

```jsx
export function Clock ({ delay = 1000 }) {
  const [count, setCount] = useState(0)

  useInterval(() => {
    setCount(count => count + 1)
  }, delay)

  const renderFunc = ({ loading, error, data }) => {
    if (loading) return <p>Loading...</p>
    if (error) return <p>{error.name}: {error.message}</p>
    return (<pre>Tada:{JSON.stringify(data)}</pre>)
  }
  return (
    <div>
      <h4>Counter: {count}</h4>
      <Fetch
        url='https://time.qcic.n.imetrical.com/'
        render={renderFunc}
      />
    </div>
  )
}
```
